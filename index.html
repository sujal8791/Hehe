import tkinter as tk import math import time # Function to create the animated flower def create_flower(canvas): # Center of the flower center_x = 250 center_y = 250 petal_length = 60 petal_width = 20 # Number of petals num_petals = 6 petal_angle = 360 / num_petals # Draw the flower's petals petals = [] for i in range(num_petals): angle = math.radians(i * petal_angle) petal_x1 = center_x + petal_length * math.cos(angle) petal_y1 = center_y + petal_length * math.sin(angle) petal_x2 = center_x + petal_width * math.cos(angle + math.radians(petal_angle / 2)) petal_y2 = center_y + petal_width * math.sin(angle + math.radians(petal_angle / 2)) petals.append(canvas.create_line(center_x, center_y, petal_x1, petal_y1, width=3, fill="yellow")) petals.append(canvas.create_line(center_x, center_y, petal_x2, petal_y2, width=3, fill="yellow")) # Create the center of the flower canvas.create_oval(center_x-15, center_y-15, center_x+15, center_y+15, fill="orange", outline="orange") # Function to show second slide with animation def show_second_slide(frame): # Clear the current slide for widget in frame.winfo_children(): widget.destroy() # Add a paragraph about waterfalls label = tk.Label(frame, text="How beautiful waterfalls are.\nThey are a sight of natural beauty,\noften creating a sense of tranquility and wonder.", font=("Arial", 14), justify="center") label.pack(pady=20) # Create canvas for animated flower canvas = tk.Canvas(frame, width=500, height=500, bg="black") canvas.pack() # Animate the flower (just a simple one-time drawing here) create_flower(canvas) # Adding some simple animations (a delay before showing the flower) time.sleep(0.5) # Wait before showing the flower canvas.after(500, lambda: canvas.update()) # Function to show the first slide def show_first_slide(root): frame = tk.Frame(root, width=500, height=500, bg="skyblue") frame.pack() # Add text on the first slide label = tk.Label(frame, text="HELLO THERE\nIM STAR", font=("Arial", 30, "bold"), fg="black") label.pack(pady=150) # Button to open the second slide open_button = tk.Button(frame, text="Open this", font=("Arial", 14), command=lambda: show_second_slide(frame)) open_button.pack(pady=20) # Set up the main window root = tk.Tk() root.title("Animated Flower Slide Show") root.geometry("500x500") # Show the first slide show_first_slide(root) # Start the Tkinter event loop root.mainloop()
